class Trie {

    public void insert(String word) {
        TriNode node = root;
        for (char c : word.toCharArray()) {
            if (node.nextNode[ c - 'a'] == null) {
                node.nextNode[c - 'a'] = new TriNode();
            }
            node = node.nextNode[c - 'a'];
            node.prefix++;
        }
        node.count++;
    }

    public boolean search(String word) {
        TriNode node = root;
        for (char c : word.toCharArray()) {
            if (node.nextNode[c-'a']==null) {
                return false;
            }
            node = node.nextNode[c-'a'];
        }
        return node.count > 0;
    }

            node = node.nextNode[c-'a'];
