class Solution {
    private static final int N = 26;
    public boolean isAnagram(String s, String t) {
        int[] dict = new int[N];
        for (char c : s.toCharArray()) {
            dict[c-'a'] += 1;
        }
        for (char c: t.toCharArray()) {
            dict[c-'a'] -= 1;
        }
        for (int i = 0; i < N; i++) {
            if(dict[i] != 0) {
                return false;
            }
        }
        return true;
    }
}
