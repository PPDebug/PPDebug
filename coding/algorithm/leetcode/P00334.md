public class Solution {
    public boolean increasingTriplet(int[] nums) {
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                for (int l = j+1; l < n; l++) {
                    if (nums[i] < nums[j] && nums[j] < nums[l]) {
                        return true;
                    }
                }
            }
        }
        return false;
    }
}
