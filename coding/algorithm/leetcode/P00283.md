class Solution {
    public void moveZeroes(int[] nums) {
        int len = nums.length;
        int index1 = 0;
        int index2 = 0;

        while(index2<len){
            if (nums[index2]==0 || index2<=index1){
                index2++;
                continue;
            }
            if (nums[index1]==0){
                nums[index1++] = nums[index2];
                nums[index2] = 0;
            } else {
                index1++;
            }
        }
    }

}
